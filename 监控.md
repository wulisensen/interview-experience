# 运行时错误监控
1、全局 js 捕获。同步、异步
  ```js
  window.onerror = function() {}
  ```
2、捕获未处理的 Promise 错误（unhandledrejection）
  ```js
  window.addEventListener('unhandledrejection', (event) => {  })
  ```
3、捕获资源加载错误（如图片 / 脚本加载失败）
  ```js
  // 监听资源加载错误
window.addEventListener('error', (event) => {
  const target = event.target;
  // 过滤出资源加载错误（非JS执行错误）
  if (target instanceof HTMLImageElement || 
      target instanceof HTMLScriptElement || 
      target instanceof HTMLLinkElement) {
    const errorInfo = {
      type: 'resource_error',
      resourceType: target.tagName.toLowerCase(), // 资源类型：img/script/link
      resourceUrl: target.src || target.href, // 资源URL
      timestamp: Date.now(),
      url: window.location.href,
    };
    reportError(errorInfo);
  }
}, true); // 注意：第三个参数为true，捕获阶段监听
  ```
4、react 处理
  ErrorBoundary 组件包裹，设置 FallbackComponent 和 onError
  React 19+：useErrorBoundary（函数组件版错误边界）
  手动处理： onClick、useEffect 中的错误 需要手动 try/catch

# 接口失败重试

```js
/**
 * 带重试的请求函数
 * @param {Function} requestFn - 原始请求函数（需返回 Promise）
 * @param {Object} options - 重试配置
 * @param {number} options.maxRetries - 最大重试次数（默认1次）
 * @param {number} options.delay - 重试延迟时间（ms，默认1000ms）
 * @param {Function} options.shouldRetry - 自定义重试条件（默认：网络错误/5xx 状态码）
 * @returns {Promise} 请求结果
 */
async function requestWithRetry(requestFn, {
  maxRetries = 1,
  delay = 1000,
  shouldRetry = (error, response) => {
    // 默认重试条件：网络错误 或 服务端5xx错误
    if (error) return true; // 网络错误（如断网、跨域、超时）
    return response?.status >= 500 && response?.status < 600; // 服务端错误
  }
} = {}) {
  let retryCount = 0;

  // 递归执行请求 + 重试
  const executeRequest = async () => {
    try {
      const response = await requestFn();
      // 检查响应是否需要重试（如5xx状态码）
      if (shouldRetry(null, response) && retryCount < maxRetries) {
        retryCount++;
        await new Promise(resolve => setTimeout(resolve, delay));
        return executeRequest();
      }
      return response;
    } catch (error) {
      // 捕获错误，判断是否重试
      if (shouldRetry(error) && retryCount < maxRetries) {
        retryCount++;
        console.log(`请求失败，${delay}ms 后重试（第${retryCount}次）`, error.message);
        await new Promise(resolve => setTimeout(resolve, delay));
        return executeRequest();
      }
      // 重试次数用尽，抛出最终错误
      throw error;
    }
  };

  return executeRequest();
}
```
